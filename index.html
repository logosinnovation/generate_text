        function setColor(type, hexColor) {

            const rgb = hexToRgb(hexColor);

            if (!rgb) return;

            

            document.getElementById(type + 'ColorHex').value = hexColor;

            document.getElementById(type + 'R').value = rgb.r;

            document.getElementById(type + 'G').value = rgb.g;

            document.getElementById(type + 'B').value = rgb.b;

            document.getElementById(type + 'ColorPreview').style.backgroundColor = hexColor;

            

            // Update selected state

            const palette = document.getElementById(type + 'ColorPalette');

            palette.querySelectorAll('.color-swatch').forEach(swatch => {

                swatch.classList.remove('selected');

                if (swatch.style.backgroundColor === hexColor || rgbToHex(...swatch.style.backgroundColor.match(/\d+/g).map(Number)) === hexColor) {

                    swatch.classList.add('selected');

                }

            });

            

            updateCanvas();

        }

                document.getElementById('formatLabel').textContent = formats[this.value].label;

                updateCanvas();

            });

        }


        function setupColorControls(type) {

            const hexInput = document.getElementById(type + 'ColorHex');

            const preview = document.getElementById(type + 'ColorPreview');

            const rInput = document.getElementById(type + 'R');

            const gInput = document.getElementById(type + 'G');

            const bInput = document.getElementById(type + 'B');


            // Hex input change

            hexInput.addEventListener('input', function() {

                const hex = this.value;

                const rgb = hexToRgb(hex);

                if (rgb) {

                    rInput.value = rgb.r;

                    gInput.value = rgb.g;

                    bInput.value = rgb.b;

                    preview.style.backgroundColor = hex;

                    updateCanvas();

                }

            });


            // RGB inputs change

            [rInput, gInput, bInput].forEach(input => {

                input.addEventListener('input', function() {

                    const r = Math.max(0, Math.min(255, parseInt(rInput.value) || 0));

                    const g = Math.max(0, Math.min(255, parseInt(gInput.value) || 0));

                    const b = Math.max(0, Math.min(255, parseInt(bInput.value) || 0));

                    

                    rInput.value = r;

                    gInput.value = g;

                    bInput.value = b;

                    

                    const hex = rgbToHex(r, g, b);

                    hexInput.value = hex;

                    preview.style.backgroundColor = hex;

                    updateCanvas();

                });

            });


            // Color preview click

            preview.addEventListener('click', function() {

                const colorPicker = document.createElement('input');

                colorPicker.type = 'color';

                colorPicker.value = hexInput.value;

                colorPicker.addEventListener('change', function() {

                    hexInput.value = this.value;

                    const rgb = hexToRgb(this.value);

                    if (rgb) {

                        rInput.value = rgb.r;

                        gInput.value = rgb.g;

                        bInput.value = rgb.b;

                        preview.style.backgroundColor = this.value;

                        updateCanvas();

                    }

                });

                colorPicker.click();

            });

        }


        function updateCanvas() {

            const format = document.getElementById('exportFormat').value;

            const { width, height } = formats[format];

            

            canvas.width = width;

            canvas.height = height;


            // Get values

            const text = document.getElementById('textContent').value;

            const fontFamily = document.getElementById('fontFamily').value;

            const fontSize = parseInt(document.getElementById('fontSize').value);

            const textAlign = document.getElementById('textAlign').value;

            

            const bgR = parseInt(document.getElementById('bgR').value);

            const bgG = parseInt(document.getElementById('bgG').value);

            const bgB = parseInt(document.getElementById('bgB').value);

            

            const textR = parseInt(document.getElementById('textR').value);

            const textG = parseInt(document.getElementById('textG').value);

            const textB = parseInt(document.getElementById('textB').value);


            // Clear and fill background

            ctx.fillStyle = `rgb(${bgR}, ${bgG}, ${bgB})`;

            ctx.fillRect(0, 0, width, height);


            // Set text properties

            ctx.fillStyle = `rgb(${textR}, ${textG}, ${textB})`;

            ctx.font = `${fontSize}px "${fontFamily}", sans-serif`;

            ctx.textBaseline = 'middle';


            // Handle text wrapping and alignment

            const maxWidth = width * 0.9; // Leave 10% padding

            const lineHeight = fontSize * 1.2;

            const words = text.split(' ');

            const lines = [];

            let currentLine = '';


            // Create wrapped lines

            for (let word of words) {

                const testLine = currentLine + (currentLine ? ' ' : '') + word;

                const metrics = ctx.measureText(testLine);

                

                if (metrics.width > maxWidth && currentLine) {

                    lines.push(currentLine);

                    currentLine = word;

                } else {

                    currentLine = testLine;

                }

            }

            if (currentLine) lines.push(currentLine);


            // Calculate total text height

            const totalHeight = lines.length * lineHeight;

            const startY = (height - totalHeight) / 2 + lineHeight / 2;


            // Draw each line with proper alignment

            lines.forEach((line, index) => {

                const y = startY + (index * lineHeight);

                let x;


                // Calculate x position based on alignment, but keep text block centered

                const textWidth = ctx.measureText(line).width;

                const textBlockCenter = width / 2;


                switch (textAlign) {

                    case 'left':

                        x = textBlockCenter - (maxWidth / 2);

                        ctx.textAlign = 'left';

                        break;

                    case 'right':

                        x = textBlockCenter + (maxWidth / 2);

                        ctx.textAlign = 'right';

                        break;

                    default: // center

                        x = textBlockCenter;

                        ctx.textAlign = 'center';

                }


                ctx.fillText(line, x, y);

            });

        }


        function exportImage(format) {

            try {

                const formatName = document.getElementById('exportFormat').value;

                const mimeType = format === 'png' ? 'image/png' : 'image/jpeg';

                const quality = format === 'jpeg' ? 0.95 : undefined;

                

                // Create the image data

                canvas.toBlob(function(blob) {

                    if (!blob) {

                        alert('Failed to generate image. Please try again.');

                        return;

                    }

                    

                    // Create download link

                    const url = URL.createObjectURL(blob);

                    const link = document.createElement('a');

                    link.href = url;

                    link.download = `social-media-text-${formatName.replace('x', 'x')}.${format}`;

                    

                    // Trigger download

                    document.body.appendChild(link);

                    link.click();

                    document.body.removeChild(link);

                    

                    // Clean up

                    setTimeout(() => URL.revokeObjectURL(url), 100);

                    

                }, mimeType, quality);

                

            } catch (error) {

                console.error('Export error:', error);

                

                // Fallback method

                try {

                    const formatName = document.getElementById('exportFormat').value;

                    const mimeType = format === 'png' ? 'image/png' : 'image/jpeg';

                    const dataURL = canvas.toDataURL(mimeType, 0.95);

                    

                    const link = document.createElement('a');

                    link.download = `social-media-text-${formatName}.${format}`;

                    link.href = dataURL;

                    

                    document.body.appendChild(link);

                    link.click();

                    document.body.removeChild(link);

                    

                } catch (fallbackError) {

                    console.error('Fallback export error:', fallbackError);

                    alert('Export failed. Please try right-clicking the canvas and selecting "Save image as..."');

                }

            }

        }


        function hexToRgb(hex) {

            const result = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex);

            return result ? {

                r: parseInt(result[1], 16),

                g: parseInt(result[2], 16),

                b: parseInt(result[3], 16)

            } : null;

        }


        function rgbToHex(r, g, b) {

            return "#" + ((1 << 24) + (r << 16) + (g << 8) + b).toString(16).slice(1);

        }


        async function copyToClipboard() {

            try {

                const blob = await new Promise(resolve => {

                    canvas.toBlob(resolve, 'image/png');

                });

                

                if (navigator.clipboard && window.ClipboardItem) {

                    await navigator.clipboard.write([

                        new ClipboardItem({

                            'image/png': blob

                        })

                    ]);

                    

                    // Show success feedback

                    const button = event.target;

                    const originalText = button.textContent;

                    button.textContent = '✅ Copied!';

                    button.style.background = 'linear-gradient(135deg, #28a745 0%, #20c997 100%)';

                    

                    setTimeout(() => {

                        button.textContent = originalText;

                        button.style.background = 'linear-gradient(135deg, #28a745 0%, #20c997 100%)';

                    }, 2000);

                    

                } else {

                    throw new Error('Clipboard API not supported');

                }

            } catch (error) {

                console.error('Copy to clipboard failed:', error);

                alert('Copy to clipboard not supported in this browser. Use the export buttons instead.');

            }

        }

    </script>
